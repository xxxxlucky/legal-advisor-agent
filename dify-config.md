 节点 7：Code 2 — 解析回复 + 安全检测
══════════════════════════════════════

──── 输入变量 ────

──── 代码 ────

import json

DANGER_KEYWORDS = [
    "杀人", "自杀", "暴力", "诈骗", "伪造",
    "规避法律", "钻空子", "非法", "行贿",
    "销毁证据", "作伪证"
]

def main(llm1_text: str) -> dict:
    try:
        data = json.loads(llm1_text)
    except:
        return {
            "reply": llm1_text,
            "is_safe": True,
            "need_lawyer": False,
            "summary": "",
            "urgency": "中",
            "legal_area": ""
        }

    reply = data.get("reply", "")
    need = data.get("need_lawyer", False)
    summary = data.get("summary", "")
    urgency = data.get("urgency", "中")
    legal_area = data.get("legal_area", "")

    is_safe = True
    reply_lower = reply.lower()
    for kw in DANGER_KEYWORDS:
        if kw in reply_lower:
            is_safe = False
            break

    return {
        "reply": reply,
        "is_safe": is_safe,
        "need_lawyer": need,
        "summary": summary,
        "urgency": urgency,
        "legal_area": legal_area
    }
